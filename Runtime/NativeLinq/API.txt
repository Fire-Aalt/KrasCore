// Used interface
public interface IPredicate<T> where T : unmanaged
{
    bool Match(in T value);
}

public interface ISelector<TSource, TResult>
    where TSource : unmanaged
    where TResult : unmanaged
{
    TResult Select(in TSource value);
}

// User defined predicates
public struct GreaterThan : IPredicate<int>
{
    public int Threshold;

    public bool Match(in int value)
    {
        return value > Threshold;
    }
}

// User-defined accumulation support.
public struct CustomValue
{
    public int Value;
}

public struct CustomValueAccumulator : INativeAccumulator<CustomValue>
{
    public CustomValue Add(in CustomValue total, in CustomValue value)
    {
        return new CustomValue { Value = total.Value + value.Value };
    }

    public CustomValue Divide(in CustomValue total, int count)
    {
        return new CustomValue { Value = total.Value / count };
    }
}

public static class CustomValueNativeLinqExtensions
{
    public static CustomValue Sum<TEnumerator>(this NativeQuery<CustomValue, TEnumerator> source)
        where TEnumerator : unmanaged, IEnumerator<CustomValue>
    {
        return source.Sum<CustomValue, TEnumerator, CustomValueAccumulator>();
    }

    public static CustomValue Average<TEnumerator>(this NativeQuery<CustomValue, TEnumerator> source)
        where TEnumerator : unmanaged, IEnumerator<CustomValue>
    {
        return source.Average<CustomValue, TEnumerator, CustomValueAccumulator>();
    }
}
